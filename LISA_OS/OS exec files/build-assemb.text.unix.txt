$EXEC {'assemb'--perform an assembly
       %0 -- pathname of the unit to assemble
       %1 -- (optional) pathname of the resulting object.  Default name is %0
       %2 -- (optional) segment name for the resulting object file}
$IF %0='' THEN
   $WRITE 'FILE TO ASSEMBLE? '
   $READLN %0
   $IF %1='' THEN
      $WRITE "NAME FOR OBJECT FILE [<CR> FOR %0]? "
      $READLN %1
   $ENDIF
   $IF %2='' THEN
      $WRITE "NAME OF CODE SEGMENT [<CR> TO SKIP CHANGESEG]? "
      $READLN %2
   $ENDIF
$ENDIF
$DEFAULT %1 TO %0
A{assemble}%0{source file}
{no listing file}
%1{object file}
$IF %2<>'' THEN
  RCHANGESEG
  %1
  Y%2
$ENDIF
$ENDEXEC
