;------------------------------------------------------------------------

; Alert Manager Call Proc

;------------------------------------------------------------------------

    .PROC   AlertProcCaller

    .DEF    CallProc
;   procedure CallProc(proc: procPtr);

;------------------------------------------------------------------------

CallProc:

        MOVE.L  4(SP),A0                ; Load address of proc to be called
        MOVE.L  (SP)+,(SP)              ; Put return address on stack
        JMP     (A0)                    ; Jump to proc
    .END
