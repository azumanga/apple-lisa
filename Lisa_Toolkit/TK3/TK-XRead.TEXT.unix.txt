EXEC(Ln,Ch,Flag)
    {converts lines beginning with '$' to just '$';
     ignores lines that begin with ';'
     if a line begins with '~', the '~' is stripped off and the rest of the line is returned.}
REPEAT
    SET Flag TO ''
    READLN (file) Ln
    SET Ch TO COPY(Ln,1,1)
    IF Ch = '$' THEN
        SET Ln TO '$'
    ELSEIF Ch = ';' THEN
        SET Flag TO 'T'
    ELSEIF Ch = '~' THEN
        SET Ln TO COPY(Ln,2,Length(Ln)-1)
    ENDIF
UNTIL Flag = ''

IF Ln = 'EOF' THEN
    SET Ln TO '$'
ENDIF

RETURN Ln
ENDEXEC
